#include
using namespace std;
#define MAX 5
int stack[MAX];
int top = -1;
void push(int x)
{ if (top == MAX-1)
{
cout << "Stack Overflow" << endl;
} else {
top++;
stack[top] = x;
}
}
int pop(void)
{
int x;
if (top == -1)
{
cout << "Stack Underflow" << endl;
return(-99999);
}
else {
x = stack[top];
top--;
return(x);
}
}

Circular Queue
#include
using namespace std;
#define MAX 10
char queue[MAX];
int front = -1, rear = -1;
void encqueue(char x)
{ if ((front == 0) && (rear == MAX - 1) || (front == rear + 1))
{ cout << "Queue is FULL" << endl;
return;
} else if (front == -1)
{ front = 0;
rear = 0;
} else if (rear == MAX - 1) rear = 0;
else rear++;
queue[rear] = x;
}
char decqueue()
{
int x;
if (front == -1)
{
cout << "The Queue is EMPTY!" << endl;
return('@');
}
else {
x = queue[front];
if (front == MAX - 1) front = 0;
else if (front == rear)
{
front = -1;
rear = -1;
} else front++;
return (x);
}
}
int main()
{
encqueue('H');
encqueue('O');
encqueue('M');
encqueue('z');
cout << decqueue() << endl;
cout << decqueue() << endl;
cout << decqueue() << endl;
}
